73 template <class T>
74 image<T>::image(const int width, const int height, const bool init) {
75  w = width;
76  h = height;
77  data = new T[w * h]; // allocate space for image data
78  access = new T*[h]; // allocate space for row pointers
79 
80  // initialize row pointers
81  for (int i = 0; i < h; i++)
82  access[i] = data + (i * w);
83 
84  if (init)
85  memset(data, 0, w * h * sizeof(T));
86 }

95 template <class T>
96 void image<T>::init(const T &val)
97 {
98  T *ptr = imPtr(this, 0, 0);
99  T *end = imPtr(this, w-1, h-1);
100  while (ptr <= end)
101  *ptr++ = val;
